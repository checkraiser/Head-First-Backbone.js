Backbone.sync = function(method, model, options) {
  var resp;
  switch (method) {
    case "create": /* ... */ break;
    case "read":   /* ... */ break;
    case "update": /* ... */ break;
    case "delete": /* ... */ break;
  }
  if (resp) {
    options.success(model, resp);
  } else {
    options.error((model, resp);
  }
};
